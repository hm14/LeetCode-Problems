# Problem Prompt # Write a program that outputs the string representation of numbers from 1 to n. # But for multiples of three it should output “Fizz” instead of the number and for the multiples of five output “Buzz”. # For numbers which are multiples of both three and five output “FizzBuzz”. def fizzBuzz(n): output = [] if n < 1: return output else: for num in range(1, n+1): if num % 15 == 0: output.append('FizzBuzz') elif num % 3 == 0: output.append('Fizz') elif num % 5 == 0: output.append('Buzz') else: output.append(str(num)) return output # Test cases for i in range(1, 20): print fizzBuzz(i)
